levels(adae$FMQ01SC) <- c("BROAD", "NARROW")

adae$FMQ01NAM <- factor(adae$FMQ01NAM, levels = c(unique(adae$FMQ01NAM), "Erectile Dysfunction", "Gynecomastia"))

# Pre-processing --------------------------------------------
adae <- adae |>
  filter(
    SAFFL == "Y",
    SEX == "M",
    FMQ01SC == "BROAD"
  )

adsl <- adsl |>
  filter(SAFFL == "Y") # safety population
